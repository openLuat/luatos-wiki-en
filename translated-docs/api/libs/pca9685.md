**Example**

```lua
--Note: due to the use of sys.wait() all apis need to be used in the coroutine
-- Usage Examples
PROJECT = "pca9685"
VERSION = "1.0.0"


sys = require("sys")
pca9685=require("pca9685")

local i2c_id = 1
sys.taskInit(
    function()
        i2c.setup(i2c_id,i2c.SLOW)
        sys.wait(1000)
        pca9685.Init(i2c_id,60)
        sys.wait(1000)
        local i=0
        while true do
            pca9685.setPWM(i2c_id,0,i)
            if i >= 100 then
                i=0
            end
            i=i+10
            sys.wait(2000)
        end
    end
)

sys.run()

```
